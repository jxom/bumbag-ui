# Pagination

## Import

```jsx
import { Pagination } from "bumbag";
```

## Usage

```function-live
function Example() {
  const [currentPage, setCurrentPage] = React.useState(1);

  return (
    <Pagination
      currentPage={currentPage}
      onChangePage={page => setCurrentPage(page)}
      numberOfPages={10}
    />
  )
}
```

### Custom labels

```function-live
function Example() {
  const [currentPage, setCurrentPage] = React.useState(1);

  return (
    <Pagination
      nextText="Next page"
      previousText="Previous page"
      prepositionText="out of"
      currentPage={currentPage}
      onChangePage={page => setCurrentPage(page)}
      numberOfPages={10}
    />
  )
}
```
