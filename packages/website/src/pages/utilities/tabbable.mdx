# Tabbable

## Import

```jsx
import { Tabbable } from 'fannypack';
```

## Usage

```function-live
function Example() {
  const onClick = () => alert('clicked');

  return (
    <div>
      <Tabbable use="div" onClick={onClick}>
        Tabbable
      </Tabbable>
      <Tabbable use="div" onClick={onClick} disabled>
        Disabled
      </Tabbable>
      <Tabbable use="div" onClick={onClick} disabled focusable>
        Focusable
      </Tabbable>
    </div>
  )
}
```
