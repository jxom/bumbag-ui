# Popover

## Import

```jsx
import { Popover } from 'fannypack';
```

## Usage

```jsx-live
<Popover.State>
  <Popover.Disclosure use={Button}>Open Popover</Popover.Disclosure>
  <Popover hasArrow>
    What was the best thing that happened to you today?
  </Popover>
</Popover.State>
```

### Placement

```jsx-live
<Popover.State placement="top">
  <Popover.Disclosure use={Button}>Open Popover</Popover.Disclosure>
  <Popover hasArrow>
    What was the best thing that happened to you today?
  </Popover>
</Popover.State>
```

### Title

```jsx-live
<Popover.State>
  <Popover.Disclosure use={Button}>Open Popover</Popover.Disclosure>
  <Popover title="Attention!">
    What was the best thing that happened to you today?
  </Popover>
</Popover.State>
```

### Footer

```jsx-live
<Popover.State>
  <Popover.Disclosure use={Button}>Open Popover</Popover.Disclosure>
  <Popover footer="This is a footer">
    What was the best thing that happened to you today?
  </Popover>
</Popover.State>
```

```jsx-live
<Popover.State>
  {popover => (
    <React.Fragment>
      <Popover.Disclosure use={Button}>Open Popover</Popover.Disclosure>
      <Popover footer={<Button onClick={popover.hide} size="small">Close</Button>}>
        What was the best thing that happened to you today?
      </Popover>
    </React.Fragment>
  )}
</Popover.State>
```

### Action buttons

```jsx-live
<Popover.State>
  <Popover.Disclosure use={Button}>Open Popover</Popover.Disclosure>
  <Popover showActionButtons actionButtonsProps={{ submitText: 'Save' }}>
    What was the best thing that happened to you today?
  </Popover>
</Popover.State>
```

### Close within popover

```jsx-live
<Popover.State>
  {popover => (
    <React.Fragment>
      <Popover.Disclosure use={Button}>Open Popover</Popover.Disclosure>
      <Popover hasArrow>
        What was the best thing that happened to you today?
        <Button onClick={popover.hide}>
          Close
        </Button>
      </Popover>
    </React.Fragment>
  )}
</Popover.State>
```

Or you can display a close button:

```jsx-live
<Popover.State>
  <Popover.Disclosure use={Button}>Open Popover</Popover.Disclosure>
  <Popover title="Attention!" showCloseButton>
    What was the best thing that happened to you today?
  </Popover>
</Popover.State>
```

### Render within a portal

```jsx-live
<Popover.State>
  <Popover.Disclosure use={Button}>Open Popover</Popover.Disclosure>
  <Popover usePortal>
    What was the best thing that happened to you today?
  </Popover>
</Popover.State>
```

### Setting initial focus

```function-live
function Example() {
  const buttonRef = React.useRef();

  return (
    <Popover.State>
      <Popover.Disclosure use={Button}>Open Popover</Popover.Disclosure>
      <Popover unstable_initialFocusRef={buttonRef}>
        What was the best thing that happened to you today?
        <Button>Button 1</Button>
        <Button ref={buttonRef}>Button 2</Button>
      </Popover>
    </Popover.State>
  )
}
```

### Setting final focus

```function-live
function Example() {
  const buttonRef = React.useRef();

  return (
    <Popover.State>
      <Popover.Disclosure use={Button}>Open Popover</Popover.Disclosure>
      <Button ref={buttonRef}>Button 2</Button>
      <Popover unstable_finalFocusRef={buttonRef}>
        What was the best thing that happened to you today?
      </Popover>
    </Popover.State>
  )
}
```

### Restricting closure

```jsx-live
<Popover.State>
  {popover => (
    <React.Fragment>
      <Popover.Disclosure use={Button}>Open Popover</Popover.Disclosure>
      <Popover hideOnEsc={false} hideOnClickOutside={false}>
        What was the best thing that happened to you today?
        <Button onClick={popover.hide}>
          Close
        </Button>
      </Popover>
    </React.Fragment>
  )}
</Popover.State>
```
